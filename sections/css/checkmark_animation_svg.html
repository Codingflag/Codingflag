html
<div class="container">
	<svg id="checkmark" 
		version="1.1" 
		xmlns="http://www.w3.org/2000/svg" 
		xmlns:xlink="http://www.w3.org/1999/xlink" 
		width="512" 
		height="512" 
		viewBox="0 0 512 512">
			<g id="icomoon-ignore"></g>
			<path id="checkmarkpath" d="M432 64l-240 240-112-112-80 80 192 192 320-320z"></path>

	<button class="toggle-btn" onclick="checkmarkpath.classList.toggle('checked')">Toggle</button>
</div>
