Giphy API Demo

POSTED ON APRIL 4, 2017


Parameters

  • q - search query term or phrase
  • limit - (optional) number of results to return, maximum 100. Default 25.
  • offset - (optional) results offset, defaults to 0.
  • rating - (optional) limit results to those rated (y,g, pg, pg-13 or r).
  • fmt - (optional) return results in html or json format (useful for viewing responses as GIFs to debug/test)

JavaScript
var gifs = document.getElementById('gifs');

function getAjaxObject(url){
  var ajax = new XMLHttpRequest();
  ajax.open("GET",url,true);
  ajax.send();
  return ajax;
}
function getGifs(){
  var searchTerm = document.getElementById('search').value;
  gifs.innerHTML = 'Searching....';
  var ajax = getAjaxObject('http://api.giphy.com/v1/gifs/search?q='+searchTerm+'&limit=10&api_key=dc6zaTOxFJmzC');
  ajax.onreadystatechange = function(){
    if(this.readyState === 4 && this.status === 200){
      var response = this.responseText;
      var gifObj = JSON.parse(response);
      console.log(gifObj);
      var gif,htmCode ='';
      for(gif in gifObj.data){
        htmCode += '<p><img src="'+gifObj.data[gif].images.downsized.url+'"/></p>';
      }
      gifs.innerHTML = htmCode;
    }
  }
}
