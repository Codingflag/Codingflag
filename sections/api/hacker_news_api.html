JavaScript
var news = document.getElementById('news');

function getAjaxObject(url){
  var ajax = new XMLHttpRequest();
  ajax.open("GET",url,true);
  ajax.send();
  return ajax;
}
function getNews(){
  var searchTerm = document.getElementById('search').value;
  news.innerHTML = 'Searching....';
  var ajax = getAjaxObject('https://hn.algolia.com/api/v1/search_by_date?query='+searchTerm);
  ajax.onreadystatechange = function(){
    if(this.readyState === 4 && this.status === 200){
      var response = this.responseText;
      var newsObj = JSON.parse(response);
      var News,htmCode ='';
      for(News in newsObj.hits){
        if(newsObj.hits[News].story_title == null) { continue; }
        htmCode += '<div><p class="title">'+newsObj.hits[News].story_title+'</p><p class="comment">'+newsObj.hits[News].comment_text+'</p><p><a href="'+newsObj.hits[News].story_url+'">Get more info here</a></p></div>';
      }
      news.innerHTML = htmCode;
    }
  }
}
